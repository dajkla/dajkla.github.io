易错题整理

* 易错题001
#include <stdio.h>
    main( )
    {
        char c[2] [5]={ "6934", "8254"},* p[2];
        int i, j, s=0;
        for(i=0;i<2;i++)
        p[i] = c[i];    //二维数组的每一行都是一个数组，c[i]相当于数组名，代表一个地址，可以赋值给指针数组中元素
        for( i=0;i<2;i++)
            for( j=0;p[i][j]>0 && p[i][j]<='9';j+=2)//指针数组p[i]存放的是地址，相当于一个数组名，后再加[]构成p[][]相当于访问数组元素
                                                    //字符串是一个以'\0'结尾的数组，'\0'的数值==0
            {
                s = 10 * s + p[i][j] - '0';         //int类型和char类型转换char转int：char -'0';int转char：int +'0'     
            }                                  
            
        printf("%d\n",s);
        //通过本题暂且粗鄙理解：地址<=>数组名

    }
